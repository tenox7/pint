# Windows NT 3.5 ARC loader (ARM32) on a real Raspberry Pi 2 - HDMI console.
#
# The GPU firmware (bootcode.bin -> start.elf) reads this file, then loads the
# kernel image into RAM and runs it. Our loader is a raw ARM binary linked at the
# firmware's native 32-bit load address (0x8000), so no kernel_address override is
# needed - it is loaded exactly where it expects to run.

arm_64bit=0
kernel=arcfw.bin

# Stage the FAT disk image in RAM, exactly as an initrd is loaded for Linux: the GPU
# firmware reads ramdisk.img off this (FAT32) boot partition and places it at 0x00800000
# (8 MiB) before entering the loader at 0x8000. The loader serves it as its Arc disk, so
# no in-loader SD driver is needed. NOTE: this directive uses a SPACE, not '=', and only
# takes effect with a loader built RAMDISK_INITRAMFS=1 (see ARM32/build/Makefile); a
# default (embedded-blob) loader ignores the staged image harmlessly. 8 MiB sits above
# the loader+stack and below the device tree at 0x02000000.
initramfs ramdisk.img 0x00800000

# Pin the device tree well clear of the loader. The loader image + stack span
# ~0x8000-0x21800; default DTB placement varies by firmware version and can land in
# that range, which our stack would then clobber (a silent crash in C - the loader
# never reads the DTB, so moving it to 32 MB is purely defensive and free).
device_tree_address=0x2000000

# Keep the serial console (PL011 / GPIO14-15) alive: enable_uart=1 also pins the
# core clock so the 115200 baud rate the loader programs does not drift.
enable_uart=1

# Force HDMI on even if the monitor is not detected at boot, and use the full
# HDMI signal (not DVI) so audio-capable displays still show video.
hdmi_force_hotplug=1
hdmi_drive=2

# If the screen stays blank on a particular monitor, uncomment this conservative
# mode and re-copy the file (it forces a safe 640x480-ish HDMI mode):
# hdmi_safe=1
